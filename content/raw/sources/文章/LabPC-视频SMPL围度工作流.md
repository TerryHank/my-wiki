```python
video = load_video(path)

frames = sample_keyframes(video, num_frames=15)

observations = []
init_params = []

for frame in frames:
    mask = segment_person(frame)
    keypoints = detect_keypoints(frame)

    theta_i, beta_i, cam_i = pixie_predict(frame)

    observations.append({
        "frame": frame,
        "mask": mask,
        "keypoints": keypoints,
    })
    init_params.append({
        "theta": theta_i,
        "beta": beta_i,
        "cam": cam_i,
    })

beta_init = robust_init([x["beta"] for x in init_params])   # median / weighted mean

beta = make_trainable(beta_init)
thetas = make_trainable([x["theta"] for x in init_params])
cams = make_trainable([x["cam"] for x in init_params])

for step in range(num_iters):
    loss = 0.0

    for i in range(len(frames)):
        verts_i, joints_i = smpl_forward(beta, thetas[i])

        proj_joints = project(joints_i, cams[i])
        render_mask = render_silhouette(verts_i, cams[i])

        loss += keypoint_loss(proj_joints, observations[i]["keypoints"])
        loss += silhouette_loss(render_mask, observations[i]["mask"])

        if i > 0:
            loss += temporal_loss(thetas[i], thetas[i-1], cams[i], cams[i-1])

    loss += shape_prior(beta)
    loss += pose_prior(thetas)

    optimize(loss)

final_mesh = smpl_canonical_mesh(beta)

neck_c = compute_circumference(final_mesh, neck_plane)
waist_c = compute_circumference(final_mesh, waist_plane)
hip_c = compute_circumference(final_mesh, hip_plane)
```
