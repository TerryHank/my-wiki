# 矩阵验收工作台重构实现计划

> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 将当前验收系统重构为 `A` 方案矩阵工作台：左侧固定 `2 x 2` 四宫格、右侧按总点数自动生成的蛇形密集矩阵、默认成功、失败点红色、悬停显示点号/坐标/当前场景备注、备注按场景保留历史。

**架构：** 保留 `VitePress + Node.js + SQLite` 架构不变，但彻底替换旧的“多次数测试方块”模型。前后端共享一套蛇形矩阵算法和稀疏异常记录模型，前端以左侧四宫格工作台和右侧密集矩阵总览为中心重建交互；后端改为“默认成功、只存失败与备注”的场景点位记录。

**技术栈：** VitePress、Vue 3、TypeScript、Node.js、SQLite、Vitest、XLSX

```ts
export function getMatrixDimensions(pointCount: number) {
  const columnCount = Math.ceil(Math.sqrt(pointCount))
  const rowCount = Math.ceil(pointCount / columnCount)
  return { columnCount, rowCount }
}

export function getPointCoordinate(pointId: number, columnCount: number) {
  const zeroBased = pointId - 1
  const y = Math.floor(zeroBased / columnCount)
  const offset = zeroBased % columnCount
  const x = y % 2 === 0 ? offset : columnCount - 1 - offset
  return { x, y }
}

export function getPointIdAtCoordinate(x: number, y: number, pointCount: number, columnCount: number) {
  const base = y * columnCount
  const offset = y % 2 === 0 ? x : columnCount - 1 - x
  const pointId = base + offset + 1
  return pointId > pointCount ? null : pointId
}
```

- [ ] **步骤 4：补齐分组与坐标格式化工具**

```ts
export function getGroupIndex(pointId: number, groupSize = 4) {
  return Math.floor((pointId - 1) / groupSize) + 1
}

export function getGroupPointIdsForPoint(pointId: number, pointCount: number, groupSize = 4) {
  const start = Math.floor((pointId - 1) / groupSize) * groupSize + 1
  const end = Math.min(pointCount, start + groupSize - 1)
  return Array.from({ length: end - start + 1 }, (_, index) => start + index)
}

export function formatAxisValue(value: number) {
  return value.toString().padStart(2, '0')
}
```

```ts
export function getScenePointState(pointId: number, sceneId: number, recordMap: Map<string, ScenePointRecord>) {
  const record = recordMap.get(`${pointId}:${sceneId}`)
  return {
    status: record?.isFailed ? 'failed' : 'success',
    note: record?.note ?? '',
    updatedAt: record?.updatedAt ?? ''
  }
}

export function calculateSceneStats(pointCount: number, scenes: Scene[], records: ScenePointRecord[]) {
  return scenes.map((scene) => {
    const sceneRecords = records.filter((record) => record.sceneId === scene.id)
    const failureCount = sceneRecords.filter((record) => record.isFailed).length
    const notedCount = sceneRecords.filter((record) => record.note.trim().length > 0).length
    return {
      sceneId: scene.id,
      sceneName: scene.name,
      failureCount,
      successCount: pointCount - failureCount,
      notedCount
    }
  })
}
```

```ts
function upsertScenePointRecord(input: { pointId: number; sceneId: number; isFailed: boolean; note: string }) {
  const trimmedNote = input.note.trim()
  if (!input.isFailed && trimmedNote === '') {
    database.prepare('delete from scene_point_records where point_id = ? and scene_id = ?').run(input.pointId, input.sceneId)
    return
  }

  database
    .prepare(`
      insert into scene_point_records (point_id, scene_id, is_failed, note, updated_at)
      values (?, ?, ?, ?, ?)
      on conflict(point_id, scene_id) do update set
        is_failed = excluded.is_failed,
        note = excluded.note,
        updated_at = excluded.updated_at
    `)
    .run(input.pointId, input.sceneId, input.isFailed ? 1 : 0, trimmedNote, new Date().toISOString())
}
```
